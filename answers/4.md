```js
//1
const fruits = ['Apple', 'Orange'];

console.log(allFruits)

//2
const fruits = ['Apple', 'Orange'];
const otherFruits = ['Orange', 'Plum', 'Pineapple'];

const allFruits = [...fruits, ...otherFruits].reduce((acc, curr) => {
  if (!acc.includes(curr)) {
    acc.push(curr)
  }
  return acc;
}, [])

console.log(allFruits)
```
